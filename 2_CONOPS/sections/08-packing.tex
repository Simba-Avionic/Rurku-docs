\section{Competition packing}

\guidance{Packing is its own ops stage: nothing may ``stay in someone's head
in the workshop''. Goal --- depart with a complete, signed kit.}

\subsection{Goal and success criteria}

\begin{itemize}
  \item \textbf{Goal:} \todo{vehicle + GSE + spares + documents + tools
    packed and verified}
  \item \textbf{Success:} packing list 100\%, owners signed their boxes,
    known location of every critical part
  \item \textbf{Failure:} \todo{critical part missing discovered only on-site}
\end{itemize}
