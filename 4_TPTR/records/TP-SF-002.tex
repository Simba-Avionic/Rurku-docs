\subsection*{Plan}

\paragraph{Objective}
Retest after TP-SF-001 igniter retention failure; demonstrate ignition and
nominal burn per DRM-0400; collect chamber pressure, thrust, and tank telemetry;
close ENG-002 / ENG-003.

\paragraph{Configuration}
\begin{itemize}
  \item \textbf{Propellants:} ethanol (fuel), nitrous oxide (oxidizer), nitrogen
    (pressurant).
  \item \textbf{Load cell:} 1$\times$ strain gauge, 240\,kHz, $\pm$1\,N.
  \item \textbf{Pressure transducers:}
    \begin{itemize}
      \item 4$\times$ PT5402 (250\,bar), 1\,kHz, $\pm$0.1\,bar --- 3$\times$
        chamber, 1$\times$ ethanol tank;
      \item 1$\times$ PT5402 (250\,bar), 60\,Hz, $\pm$0.5\,bar --- N\textsubscript{2}O
        tank;

        tank.
    \end{itemize}

\paragraph{Success criteria}
\begin{itemize}
  \item Successful ignition inside the combustion chamber.
  \item Chamber pressure and thrust within expected band for the planned burn
    duration (5\,s nominal).
  \item Shutdown sequence completes without critical hardware damage.
  \item Complete DAQ log and post-test photos.
\end{itemize}

\paragraph{Failure / abort criteria}
\begin{itemize}
  \item Misfire $\rightarrow$ hold; follow misfire wait and approach procedure.
  \item Loss of critical telemetry (chamber pressures, load cell) $\rightarrow$
    HOLD / abort.
  \item Sensor out of limits before ignition $\rightarrow$ HOLD / abort.
  \item Combustion outside the combustion chamber (e.g.\ ejected igniters).
  \item Hard start / overpressure $\rightarrow$ execute D-3 hot abort; safing per
    D-1 / D-2.
\end{itemize}
