\section{Context and ownership}

\guidance{Every interface has two sides. Both owners must agree before status
moves to Approved.}

\subsection{Stakeholders}

\begin{itemize}
  \item Interface owners (both sides of each \texttt{IF-*})
  \item System / project lead (arbitrates boundary disputes)
  \item Integration / test lead (uses ICD as V\&V input)
\end{itemize}

\subsection{Owner rule}

\begin{itemize}
  \item \textbf{Side A / Side B} --- design owners named on each IF sheet
  \item Changing an interface parameter requires both sides (or written waiver
    by project lead)
  \item SAD keeps the index; ICD keeps the controlled definition
\end{itemize}
